[docs]class ExtentMissingError(Exception):

    _attribute = None

    def __init__(self, attribute):
        self._attribute = attribute

    def __str__(self):
        return ERR_DOCUMENT_EXTENT_MISSING.format(type=type(self._attribute), value=self._attribute)
